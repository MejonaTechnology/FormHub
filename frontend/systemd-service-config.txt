# FormHub Frontend SystemD Service Configuration
# This file shows the proper configuration for the formhub-frontend.service

[Unit]
Description=FormHub Frontend
After=network.target

[Service]
Type=simple
User=ubuntu
WorkingDirectory=/opt/mejona/formhub-frontend
Environment=NODE_ENV=production
Environment=NEXT_PUBLIC_API_URL=http://13.127.59.135:9000/api/v1
ExecStart=/usr/bin/npm run start:production
Restart=always
RestartSec=10
StandardOutput=journal
StandardError=journal
SyslogIdentifier=formhub-frontend

[Install]
WantedBy=multi-user.target

# To apply this configuration:
# 1. Copy this content to: /etc/systemd/system/formhub-frontend.service
# 2. Run: sudo systemctl daemon-reload
# 3. Run: sudo systemctl enable formhub-frontend
# 4. Run: sudo systemctl start formhub-frontend
